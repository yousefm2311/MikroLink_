import cron from "node-cron";
import Driver from "./models/Driver.js";
import DriverLocation from "./models/DriverLocation.js";

// 🕓 كل 10 دقايق
cron.schedule("*/10 * * * *", async () => {
  try {
    const tenMinsAgo = new Date(Date.now() - 10 * 60 * 1000);

    // هات كل السواقين اللي آخر تحديث لهم أقدم من 10 دقايق
    const inactiveDrivers = await DriverLocation.find({
      updatedAt: { $lt: tenMinsAgo },
    });

    for (const loc of inactiveDrivers) {
      await Driver.findByIdAndUpdate(loc.driverId, { isOnline: false });
    }

    console.log(`⏰ تم تحويل ${inactiveDrivers.length} سائق إلى Offline`);
  } catch (err) {
    console.error("❌ خطأ في عملية التحديث التلقائي:", err.message);
  }
});
